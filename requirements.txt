# Requirements for the IFC AI Assistant Application
# -------------------------------------------------
# Install these packages using:
# pip install -r requirements.txt
# -------------------------------------------------

# Core web application framework
streamlit

# For parsing and interacting with Industry Foundation Classes (IFC) files
ifcopenshell

# For numerical operations, especially in geometric calculations
numpy

# Core LangChain framework for building the agent
langchai

# LangChain's integration with Google's Generative AI models (Gemini)
langchain-google-genai

# LangChain community components, including the Streamlit callback handler
langchain-community

# Core abstractions for LangChain (often a dependency, but good to specify)
langchain-core
